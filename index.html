📚 Пример использования

        
import requests

# Ваш API ключ
API_KEY = "ваш_ключ_здесь"

# Данные для конвертации
data = [
    {"name": "Иван", "age": 30},
    {"name": "Мария", "age": 25}
]

# Отправляем запрос
response = requests.post(
    "http://localhost:8000/api/convert",
    headers={"x-api-key": API_KEY},
    json={
        "data": data,
        "from_format": "json",
        "to_format": "csv"
    }
)

print(response.json()["result"])
# Вывод: name,age
#        Иван,30
#        Мария,25

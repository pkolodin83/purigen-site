📚 Usage Example

        
import requests

API_KEY = "your_api_key_here"
data = [{"name": "John", "age": 30}, {"name": "Maria", "age": 25}]

response = requests.post(
    "https://YOUR-API-URL.onrender.com/api/convert",
    headers={"x-api-key": API_KEY},
    json={"data": data, "from_format": "json", "to_format": "csv"}
)

print(response.json()["result"])
# Output: name,age
#         John,30
#         Maria,25
